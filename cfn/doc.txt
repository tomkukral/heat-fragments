Import ssh key
	aws --profile mi ec2 import-key-pair --key-name tkukral_yubi --public-key-material file://~/.ssh/tomas_kukral_yubi.pub

List keys
	aws --profile mi ec2 describe-key-pairs

Create stack
	aws --profile mi cloudformation create-stack --stack-name test-$(date +"%Y-%m-%d-%H-%M-%S") --template-body file://simple.yml --parameters ParameterKey=KeyName,ParameterValue=tkukral_yubi
